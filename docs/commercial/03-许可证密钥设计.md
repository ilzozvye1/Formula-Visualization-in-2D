```
PRO-{类型}-{日期}-{随机码}
```

## 四、密钥生成算法

### 4.1 核心生成器代码

```javascript
/**
 * 许可证密钥生成器
 */
class LicenseKeyGenerator {
  // 密钥类型
  static TYPES = {
    MONTHLY: 'M',    // 月卡
    YEARLY: 'Y',    // 年卡
    LIFETIME: 'L'   // 终身
  };
  
  // 有效期限（毫秒）
  static PERIODS = {
    M: 30 * 24 * 60 * 60 * 1000,
    Y: 365 * 24 * 60 * 60 * 1000,
    L: null
  };
  
  /**
   * 生成密钥
   * @param {string} type - 密钥类型 M/Y/L
   * @returns {string} 生成的密钥
   */
  static generate(type = 'Y') {
    const date = this.formatDate(new Date());
    const random = this.generateRandomCode();
    return `PRO-${type}-${date}-${random}`;
  }
  
  /**
   * 格式化日期
   */
  static formatDate(date) {
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const day = String(date.getDate()).padStart(2, '0');
    return `${year}${month}${day}`;
  }
  
  /**
   * 生成随机码
   * 格式：6位字母数字混合（避免易混淆字符）
   */
  static generateRandomCode() {
    const chars = 'abcdefghijkmnpqrstuvwxyz23456789';
    let code = '';
    for (let i = 0; i < 6; i++) {
      code += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return code;
  }
  
  /**
   * 生成校验码
   * 用于增强密钥合法性验证
   */
  static generateChecksum(key) {
    const parts = key.split('-');
    if (parts.length !== 4) return null;
    
    const input = parts[0] + parts[1] + parts[2];
    let hash = 0;
    for (let i = 0; i < input.length; i++) {
      hash = ((hash << 5) - hash) + input.charCodeAt(i);
      hash = hash & hash;
    }
    
    const chars = 'abcdefghijkmnpqrstuvwxyz23456789';
    const checksum = Math.abs(hash).toString(36).substring(0, 4);
    return checksum;
  }
  
  /**
   * 验证密钥格式
   */
  static validateFormat(key) {
    if (!key || typeof key !== 'string') return false;
    
    const parts = key.split('-');
    if (parts.length !== 4) return false;
    
    const [prefix, type, date, code] = parts;
    
    // 前缀检查
    if (prefix !== 'PRO') return false;
    
    // 类型检查
    if (!['M', 'Y', 'L'].includes(type)) return false;
    
    // 日期格式检查 (YYYYMMDD)
    if (!/^\d{8}$/.test(date)) return false;
    
    // 随机码检查 (6位，排除易混淆字符)
    if (!/^[a-z0-9]{6}$/.test(code)) return false;
    
    return true;
  }
  
  /**
   * 获取密钥信息
   */
  static getKeyInfo(key) {
    if (!this.validateFormat(key)) {
      return null;
    }
    
    const parts = key.split('-');
    const type = parts[1];
    const date = parts[2];
    
    const typeNames = {
      'M': '月卡',
      'Y': '年卡',
      'L': '终身'
    };
    
    return {
      type,
      typeName: typeNames[type],
      date,
      period: this.PERIODS[type],
      createdDate: this.parseDate(date)
    };
  }
  
  /**
   * 解析日期
   */
  static parseDate(dateStr) {
    const year = parseInt(dateStr.substring(0, 4));
    const month = parseInt(dateStr.substring(4, 6)) - 1;
    const day = parseInt(dateStr.substring(6, 8));
    return new Date(year, month, day);
  }
  
  /**
   * 检查密钥是否过期
   */
  static isExpired(key) {
    const info = this.getKeyInfo(key);
    if (!info || !info.period) return false;
    
    const expiryDate = new Date(info.createdDate.getTime() + info.period);
    return new Date() > expiryDate;
  }
}

// 使用示例
console.log(LicenseKeyGenerator.generate('Y'));  // PRO-Y-20250310-a3f9k2
console.log(LicenseKeyGenerator.validateFormat('PRO-Y-20250310-a3f9k2')); // true
```

## 五、密钥验证流程

### 5.1 验证流程图

```
用户输入密钥
    ↓
格式验证（本地）
    ↓
┌─ 格式错误 → 提示错误
│
└─ 格式正确 → 服务器验证
              ↓
        ┌─ 无效 → 提示无效
        │
        └─ 有效 → 检查过期
                  ↓
            ┌─ 已过期 → 提示过期
            │
            └─ 有效 → 激活成功
```

### 5.2 本地验证（格式检查）

```javascript
function validateLicenseFormat(key) {
  // 1. 基本检查
  if (!key || typeof key !== 'string') {
    return { valid: false, message: '密钥不能为空' };
  }
  
  // 2. 格式检查
  if (!LicenseKeyGenerator.validateFormat(key)) {
    return { valid: false, message: '密钥格式不正确' };
  }
  
  return { valid: true, message: '格式正确' };
}
```

### 5.3 服务器验证（可选）

// 密钥数据库结构（JSON格式存储）
// keys.json:
{
  "PRO-Y-20250310-a3f9k2": {
    "type": "Y",
    "created": "2025-03-10",
    "expires": "2026-03-10",

    "used": false,
    "orderId": "ORDER-12345",

// 密钥验证API
async function verifyLicense(key) {
  // 1. 格式验证
  if (!LicenseKeyGenerator.validateFormat(key)) {
    return { valid: false, reason: 'invalid_format' };
  }
  
  // 2. 查询数据库
  const keyData = await fetch(`/api/keys/${key}`).then(r => r.json());
  
  if (!keyData) {
    return { valid: false, reason: 'not_found' };
  }
  
  // 3. 检查是否已使用
  if (keyData.used && !keyData.orderId) {
    return { valid: false, reason: 'already_used' };
  }
  
  // 4. 检查过期
  if (keyData.expires && new Date(keyData.expires) < new Date()) {
    return { valid: false, reason: 'expired' };
  }
  
  return { valid: true, data: keyData };
}

### 6.1 激活流程

```

    ↓

    ↓
用户点击"激活Pro版"

    ↓

    ↓
┌─ 验证失败 → 提示错误
│

              ↓
         解锁Pro功能
              ↓
         刷新界面，显示Pro标识
```

### 6.2 激活实现代码

```javascript
async function activatePro(licenseKey) {
  // 1. 验证密钥格式
  if (!LicenseKeyGenerator.validateFormat(licenseKey)) {
    return { success: false, message: '密钥格式不正确' };
  }
  
  // 2. 验证密钥有效性（可选服务器验证）
  const verifyResult = await verifyLicense(licenseKey);
  if (!verifyResult.valid) {
    return { success: false, message: `密钥无效：${verifyResult.reason}` };
  }
  
  // 3. 保存到本地存储
  localStorage.setItem('pro_license', licenseKey);
  localStorage.setItem('pro_activated', Date.now().toString());
  localStorage.setItem('pro_platform', getPlatform());
  
  // 4. 获取密钥信息并保存
  const keyInfo = LicenseKeyGenerator.getKeyInfo(licenseKey);
  localStorage.setItem('pro_type', keyInfo.type);
  localStorage.setItem('pro_expires', keyInfo.period ? 
    new Date(keyInfo.createdDate.getTime() + keyInfo.period).toISOString() : 'never');
  
  // 5. 解锁Pro功能
  enableProFeatures();
  
  // 6. 刷新界面
  showProWelcome();
  
  return { success: true, message: 'Pro版已激活！' };
}
```

### 6.3 Pro状态检查

```javascript
function checkProStatus() {
  const license = localStorage.getItem('pro_license');
  const activated = localStorage.getItem('pro_activated');
  
  if (!license || !activated) {
    return { isPro: false, reason: 'not_activated' };
  }
  
  // 检查过期
  const expires = localStorage.getItem('pro_expires');
  if (expires && expires !== 'never') {
    if (new Date(expires) < new Date()) {
      return { isPro: false, reason: 'expired', expiredDate: expires };
    }
  }
  
  return { 
    isPro: true, 
    license,
    activated: new Date(parseInt(activated)),
    expires: expires === 'never' ? null : new Date(expires)
  };
}
```

## 七、密钥管理

### 7.1 密钥池管理

```javascript
// 密钥生成与池管理
class KeyPool {
  constructor() {
    this.keys = [];
  }
  
  // 生成指定数量的密钥
  generateBatch(type, count) {
    const keys = [];
    for (let i = 0; i < count; i++) {
      const key = LicenseKeyGenerator.generate(type);
      keys.push({
        key,
        type,
        created: new Date().toISOString(),
        used: false
      });
      this.keys.push(key);
    }
    return keys;
  }
  
  // 导出密钥列表
  exportKeys() {
    return JSON.stringify(this.keys, null, 2);
  }
  
  // 导入密钥列表
  importKeys(jsonString) {
    this.keys = JSON.parse(jsonString);
  }
}
```

## 八、安全考虑

### 8.1 密钥安全原则

1. **不存储在客户端** - 密钥验证仅做格式检查，不存储敏感信息
2. **可选服务器验证** - 可选增加服务端验证增强安全性
3. **一次性使用** - 每个密钥可在多平台使用，但需记录
4. **定期更换** - 定期生成新密钥池，防止泄露

### 8.2 防破解建议
