### 5.2 平台检测逻辑

```javascript
function getPlatform() {
  const ua = navigator.userAgent;
  
  if (ua.includes('Electron')) {
    return 'desktop';
  }
  if (/iPhone|iPad|Android/i.test(ua)) {
    return 'mobile';
  }
  return 'web';
}

function isPro() {
  const license = localStorage.getItem('pro_license');
  return !!license && isValidLicense(license);
}

function canUse(feature) {
  const platform = getPlatform();
  const config = PRO_FEATURES[feature];
  
  if (!config) return true;
  
  if (typeof config === 'boolean') {
    return isPro() ? config : config;
  }
  
  if (config.free !== undefined) {
    return isPro() ? true : config.free;
  }
  
  return true;
}
```
